35pub fn transfer_hopr_token_payload(
36    token_address: Address,
37    address: Address,
38    amount: U256,
39) -> Result<TransactionRequest, HelperErrors> {
40    let transfer_function_payload = transferCall {
41        recipient: address,
42        amount,
43    }
44    .abi_encode();
45    let tx = TransactionRequest::default()
46        .with_to(token_address)
47        .with_input(transfer_function_payload);
48    Ok(tx)

51pub fn transfer_native_token_payload(
52    addresses: Vec<Address>,
53    amounts: Vec<U256>,
54) -> Result<TransactionRequest, HelperErrors> {
55    // check if two vectors have the same length
56    if addresses.len() != amounts.len() {
57        return Err(HelperErrors::MissingParameter(
58            "Addresses and amounts length mismatch".into(),
59        ));
60    }
61
62    // calculate the sum of tokens to be sent
63    let total = amounts.iter().fold(U256::ZERO, |acc, cur| acc.add(cur));
64    debug!(
65        "total amount of native tokens to be transferred {:?}",
66        total.to_string()
67    );
68
69    let calls: Vec<Call3Value> = addresses
70        .clone()
71        .into_iter()
72        .enumerate()
73        .map(|(i, addr)| Call3Value {
74            target: addr,
75            allowFailure: false,
76            value: amounts[i],
77            callData: Bytes::default(),
78        })
79        .collect::<Vec<_>>();
80    let aggregate3_value_payload = aggregate3ValueCall { calls }.abi_encode();
81    let tx = TransactionRequest::default()
82        .with_to(MULTICALL3_ADDRESS)
83        .with_input(aggregate3_value_payload)
84        .with_value(total);
85    Ok(tx)
86}

88/// Predict safe address deployed by the edge node
89pub fn edge_node_predict_safe_address(
90    hopr_node_stake_factory_address: Address,
91    hopr_channels_address: Address,
92    nonce: U256,
93    admins: Vec<Address>,
94) -> Result<Address, HelperErrors> {
95    if admins.is_empty() {
96        return Err(HelperErrors::MissingParameter(
97            "At least one admin address must be provided".into(),
98        ));
99    }
100
101    // build a new temporary admin
102    let mut temporary_admins: Vec<Address> = admins.clone();
103    temporary_admins.insert(0, MULTICALL3_ADDRESS);
104    info!(
105        "temporary_admins expands from admin from {:?} addresses to {:?}",
106        admins.len(),
107        temporary_admins.len()
108    );
109
110    // build the default permissions of capabilities
111    let default_target = U256::from_str(format!("{hopr_channels_address:?}{DEFAULT_CAPABILITY_PERMISSIONS}").as_str())
112        .map_err(|e| HelperErrors::ParseError(format!("Invalid default_target format: {e}")))?;
113    debug!("default target {:?}", default_target);
114
115    let safe_address = predict_safe_address(
116        hopr_node_stake_factory_address,
117        temporary_admins.clone(),
118        nonce.into(),
119        SAFE_COMPATIBILITYFALLBACKHANDLER_ADDRESS,
120        SAFE_SAFE_L2_ADDRESS,
121        SAFE_SAFEPROXYFACTORY_ADDRESS,
122    )?;
123    Ok(safe_address)
124}

146/// Payload for deploying safe module with target addresses and node addresses
147/// This payload assumes the threshold is one.
148/// The prediceted safe address can be obtained by calling `edge_node_predict_safe_address`
149/// The predicted module address can be obtained by calling `predictModuleAddress_1` on `hopr_node_stake_factory`
150/// providing MULTICALL3_ADDRESS, nonce.into(), safe_address, default_target
151pub fn edge_node_deploy_safe_module_with_targets_and_nodes_payload(
152    predicted_safe_address: Address,
153    predicted_module_address: Address,
154    hopr_node_stake_factory_address: Address,
155    hopr_channels_address: Address,
156    node_addresses: Vec<Address>,
157    nonce: U256,
158    admins: Vec<Address>,
159) -> Result<TransactionRequest, HelperErrors> {
160    if admins.is_empty() {
161        return Err(HelperErrors::MissingParameter(
162            "At least one admin address must be provided".into(),
163        ));
164    }
165
166    // build a new temporary admin
167    let mut temporary_admins: Vec<Address> = admins.clone();
168    temporary_admins.insert(0, MULTICALL3_ADDRESS);
169    info!(
170        "temporary_admins expands from admin from {:?} addresses to {:?}",
171        admins.len(),
172        temporary_admins.len()
173    );
174
175    // build the default permissions of capabilities
176    let default_target = U256::from_str(format!("{hopr_channels_address:?}{DEFAULT_CAPABILITY_PERMISSIONS}").as_str())
177        .map_err(|e| HelperErrors::ParseError(format!("Invalid default_target format: {e}")))?;
178    debug!("default target {:?}", default_target);
179
180    // Use multicall to deploy a safe proxy instance and a module proxy instance with multicall as an owner
181    let mut multicall_payloads: Vec<Call3> = vec![];
182    multicall_payloads.push(Call3 {
183        target: hopr_node_stake_factory_address,
184        allowFailure: false,
185        callData: cloneCall {
186            nonce,
187            defaultTarget: default_target.into(),
188            admins: temporary_admins,
189        }
190        .abi_encode()
191        .into(),
192    });
193    info!("Safe and module deployment multicall payload is created");
194
195    // if node addresses are known, include nodes to the module by safe
196    if !node_addresses.is_empty() {
197        for node in node_addresses {
198            let node_target = U256::from_str(&format!("{node:?}{DEFAULT_NODE_PERMISSIONS}"))
199                .map_err(|e| HelperErrors::ParseError(format!("Invalid node_target format: {e}")))?;
200
201            let encoded_call = includeNodeCall {
202                nodeDefaultTarget: node_target,
203            }
204            .abi_encode();
205
206            let payload = prepare_safe_tx_multicall_payload_from_owner_contract(
207                predicted_safe_address,
208                predicted_module_address,
209                predicted_safe_address,
210                encoded_call,
211            );
212
213            multicall_payloads.push(payload.to_call3());
214        }
215
216        info!("Nodes inclusion multicall payload is created");
217    } else {
218        info!("No node has been provided. Skip node inclusion action for multicall payload generation");
219    }
220
221    // renounce ownership granted to multicall so that only actual admins are included. Set the threshold.
222    let remove_owner_tx_payload = removeOwnerCall {
223        prevOwner: Address::from_str(SENTINEL_OWNERS)
224            .map_err(|e| HelperErrors::ParseError(format!("Invalid SENTINEL_OWNERS address: {e}")))?,
225        owner: MULTICALL3_ADDRESS,
226        _threshold: U256::from(1),
227    }
228    .abi_encode();
229
230    let multicall_payload_5 = prepare_safe_tx_multicall_payload_from_owner_contract(
231        predicted_safe_address,
232        predicted_safe_address,
233        predicted_safe_address,
234        remove_owner_tx_payload,
235    );
236
237    multicall_payloads.push(multicall_payload_5.to_call3());
238    info!("Admins and threshold setting multicall payload is created");
239
240    // build multicall transaction
241    let aggregate3_payload = aggregate3Call {
242        calls: multicall_payloads,
243    }
244    .abi_encode();
245    let tx = TransactionRequest::default()
246        .with_to(MULTICALL3_ADDRESS)
247        .with_input(aggregate3_payload);
248    Ok(tx)
249}

261/// Deploy a safe and a module, while sending tokens to the safe for single edge node.
262/// It's possible to only deploy a safe and a module without onboarding the node
263/// Alternatively, the node will be included in the module after deployment
264/// All the "admins" will be the nodes added to the module
265/// Returns safe proxy address and module proxy address
266pub fn edge_node_deploy_safe_module_and_maybe_include_node(
267    hopr_node_stake_factory_address: Address,
268    hopr_token_address: Address,
269    hopr_channels_address: Address,
270    nonce: U256,
271    amount: U256,
272    admins: Vec<Address>,
273    should_include_node: bool,
274) -> Result<TransactionRequest, HelperErrors> {
275    // build the default permissions of capabilities
276    let default_target = U256::from_str(format!("{hopr_channels_address:?}{DEFAULT_CAPABILITY_PERMISSIONS}").as_str())
277        .map_err(|e| HelperErrors::ParseError(format!("Invalid default_target format: {e}")))?;
278    debug!("default target {:?}", default_target);
279
280    let user_data = if should_include_node {
281        UserData {
282            functionIdentifier: DEPLOYSAFEANDMODULEANDINCLUDENODES_IDENTIFIER,
283            nonce,
284            defaultTarget: default_target.into(),
285            admins,
286        }
287        .abi_encode()[32..]
288            .to_vec()
289    } else {
290        UserData {
291            functionIdentifier: DEPLOYSAFEMODULE_FUNCTION_IDENTIFIER,
292            nonce,
293            defaultTarget: default_target.into(),
294            admins,
295        }
296        .abi_encode()[32..]
297            .to_vec()
298    };
299
300    debug!("User data for deploying safe and module: {:?}", hex::encode(&user_data));
301    let tx_payload = sendCall {
302        recipient: hopr_node_stake_factory_address,
303        amount,
304        data: user_data.into(),
305    }
306    .abi_encode();
307
308    let tx = TransactionRequest::default()
309        .with_to(hopr_token_address)
310        .with_input(tx_payload);
311    Ok(tx)
312}

314/// Set the global minimum winning probability in the HoprWinningProbabilityOracle contract
315pub fn set_winning_probability(
316    hopr_win_prob_oracle_address: Address,
317    winning_probability: f64,
318) -> Result<TransactionRequest, HelperErrors> {
319    // the winning probablity should be in range of 0.0 to 1.0 (inclusive)
320    if !(0.0..=1.0).contains(&winning_probability) {
321        return Err(HelperErrors::ParseError(
322            "Winning probability must be between 0.0 and 1.0".into(),
323        ));
324    }
325    // convert the winning probability to the format required by the contract
326    let winning_probability_val = WinningProbability::try_from(winning_probability)
327        .map_err(|_| HelperErrors::ParseError("Failed to convert winning probability to the required format".into()))?;
328
329    info!(
330        winning_probability = %winning_probability_val,
331        win_prob_f64 = %winning_probability,
332        "Setting the global minimum winning probability"
333    );
334
335    let tx_payload = setWinProbCall {
336        _newWinProb: U56::from_be_slice(&winning_probability_val.as_encoded()),
337    }
338    .abi_encode();
339
340    let tx = TransactionRequest::default()
341        .with_to(hopr_win_prob_oracle_address)
342        .with_input(tx_payload);
343    Ok(tx)
344}
